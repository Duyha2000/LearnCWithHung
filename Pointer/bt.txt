void swap(int *a, int *b) {
    int temp;
    temp = *a;
    *a = *b;
    *b = temp;
}


// Input:
void read_array(int *v, int max) {
    for (int i = 0; i < max; i++) {
        scanf("%d", v + i);
    }
}

void print_array(const int *v, int max) {
    for (int i = 0; i < max; i++) {
        printf("%d", *(v + i));
    }
}

// a1: [3] 4 7
//         ->
// a2: [5] 2 9
void swap_array(int *v1, int *v2, int max) {
    for (int i = 0; i < max; i++) {
        swap(v1 + i, v2 + i);
    }
}
